CREATE TABLESPACE TBS_BIGDATA DATAFILE 
  'H:/app/Oracle/oradata/orcl/TBS_BIGDATA.ORA' SIZE 1024M AUTOEXTEND ON NEXT 50M MAXSIZE UNLIMITED
LOGGING
ONLINE
PERMANENT
EXTENT MANAGEMENT LOCAL AUTOALLOCATE
BLOCKSIZE 8K
SEGMENT SPACE MANAGEMENT AUTO
FLASHBACK ON;


CREATE USER XZXT
  IDENTIFIED BY XZXT
  DEFAULT TABLESPACE TBS_BIGDATA
  PROFILE DEFAULT
  ACCOUNT UNLOCK;
  -- 3 Roles for XZXT
  GRANT DBA TO XZXT;
  GRANT CONNECT TO XZXT;
  GRANT RESOURCE TO XZXT;
  ALTER USER XZXT DEFAULT ROLE ALL;
  -- 4 System Privileges for XZXT
  GRANT UNLIMITED TABLESPACE TO XZXT;
  GRANT UPDATE ANY TABLE TO XZXT;
  GRANT INSERT ANY TABLE TO XZXT;
  GRANT SELECT ANY TABLE TO XZXT;
