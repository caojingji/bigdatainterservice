--创建轨迹表
CREATE TABLE XZXT.TRACK
(
    XXZJBH VARCHAR2(23 BYTE),
    ADDRESS VARCHAR2(500 BYTE),
    BASE VARCHAR2(100 BYTE),
    OBJECTTYPE VARCHAR2(10 BYTE),
    OBJECTTYPENAME VARCHAR2(20 BYTE),
    OBJECTVALUE VARCHAR2(50 BYTE),
    SOURCE VARCHAR2(50 BYTE),
    SOURCENAME VARCHAR2(50 BYTE),
    SOURCE_MD5 VARCHAR2(50 BYTE),
    TIMESTAMP VARCHAR2(50 BYTE),
    TIMESTR VARCHAR2(50 BYTE),
    J VARCHAR2(50 BYTE),
    W VARCHAR2(50 BYTE),
    DJSJ DATE               DEFAULT sysdate
)TABLESPACE TBS_BIGDATA
 PCTUSED    0
 PCTFREE    10
 INITRANS   1
 MAXTRANS   255
 STORAGE    (
             INITIAL          64K
             NEXT             1M
             MINEXTENTS       1
             MAXEXTENTS       UNLIMITED
             PCTINCREASE      0
             BUFFER_POOL      DEFAULT
            )
 LOGGING
 NOCOMPRESS
 NOCACHE
 NOPARALLEL
 MONITORING;

COMMENT ON table XZXT.TRACK is '手机轨迹表';
COMMENT ON COLUMN XZXT.TRACK.XXZJBH IS '信息主键编号';

COMMENT ON COLUMN XZXT.TRACK.ADDRESS IS '基站地址';

COMMENT ON COLUMN XZXT.TRACK.BASE IS '';

COMMENT ON COLUMN XZXT.TRACK.OBJECTTYPE IS '对象类型编号';

COMMENT ON COLUMN XZXT.TRACK.OBJECTTYPENAME IS '对象类型名称';

COMMENT ON COLUMN XZXT.TRACK.OBJECTVALUE IS '对象值';

COMMENT ON COLUMN XZXT.TRACK.SOURCE IS '资源类型编号';

COMMENT ON COLUMN XZXT.TRACK.SOURCENAME IS '资源类型名称';

COMMENT ON COLUMN XZXT.TRACK.SOURCE_MD5 IS '资源编码';
COMMENT ON COLUMN XZXT.TRACK.TIMESTAMP IS '关联时间';
COMMENT ON COLUMN XZXT.TRACK.TIMESTR IS '格式化后的关联时间';
COMMENT ON COLUMN XZXT.TRACK.J IS '经度';
COMMENT ON COLUMN XZXT.TRACK.W IS '纬度';
COMMENT ON COLUMN XZXT.TRACK.DJSJ IS '登记时间';
--创建主键
CREATE UNIQUE INDEX PK_TRACK_XXZJBH ON XZXT.TRACK
(XXZJBH)
LOGGING
TABLESPACE TBS_INDEX
PCTFREE    10
INITRANS   2
MAXTRANS   255
STORAGE    (
            INITIAL          64K
            NEXT             1M
            MINEXTENTS       1
            MAXEXTENTS       UNLIMITED

           )
